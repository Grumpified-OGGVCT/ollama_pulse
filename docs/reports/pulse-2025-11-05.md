```python
            'coding': 'qwen3-coder:480b-cloud',
            'vision': 'qwen3-vl:235b-cloud', 
            'agentic': 'glm-4.6:cloud',
            'efficient': 'minimax-m2:cloud'
        }
    
    async def code_review(self, code: str, diagram_path: str = None) -> str:
        """Combines code analysis with visual understanding"""
        tasks = []
        
        # Analyze code with the coding specialist
        tasks.append(
            self._query_model('coding', 
                f"Review this code for bugs and optimization:\n\n{code}")
        )
        
        # If there's a diagram, analyze it with VL model
        if diagram_path:
            tasks.append(
                self._query_model('vision',
                    f"Analyze this architecture diagram and suggest code improvements: {diagram_path}")
            )
        
        results = await asyncio.gather(*tasks)
        return self._synthesize_feedback(results)
    
    async def _query_model(self, model_type: str, prompt: str) -> str:
        """Generic model query method"""
        response = ollama.chat(
            model=self.models[model_type],
            messages=[{'role': 'user', 'content': prompt}]
        )
        return response['message']['content']
```

**Integration Pattern: The Model Cascade**
For complex tasks, chain models together:

```python
async def complex_feature_development(requirement: str):
    """Example of model chaining for feature development"""
    
    # Step 1: Use agentic model for planning
    plan = await query_model('agentic', 
        f"Break down this feature into tasks: {requirement}")
    
    # Step 2: Use coder for implementation
    code = await query_model('coding',
        f"Implement this plan:\n{plan}")
    
    # Step 3: Use efficient model for testing
    tests = await query_model('efficient',
        f"Generate tests for this code:\n{code}")
    
    return {'plan': plan, 'code': code, 'tests': tests}
```
