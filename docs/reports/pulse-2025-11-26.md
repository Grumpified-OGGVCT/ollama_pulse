class ContextAwareDebugger:
    def __init__(self):
        self.context_window = []
        self.max_context = 190000  # Leave buffer for responses
    
    def add_to_context(self, content):
        # Manage context window efficiently
        current_size = sum(len(str(item)) for item in self.context_window)
        new_size = current_size + len(str(content))
        
        if new_size > self.max_context:
            # Remove oldest entries, keep important error context
            self.context_window = self.context_window[-10:]  # Keep recent critical info
        
        self.context_window.append(content)
    
    def debug_issue(self, error_log, code_snippet, system_state):
        context_str = "\n".join(str(item) for item in self.context_window[-20:])
        
        prompt = f"""
        Debugging Session Context:
        {context_str}
        
        Current Issue:
        Error: {error_log}
        Code: {code_snippet} 
        System State: {system_state}
        
        Provide step-by-step analysis and fix recommendation.
        """
        
        response = ollama.chat(
            model="glm-4-6:cloud",
            messages=[{"role": "user", "content": prompt}]
        )
        
        self.add_to_context(f"Issue: {error_log}\nSolution: {response['message']['content']}")
        return response['message']['content']
