class MultiModelOrchestrator:

    def analyze_ui_issue(self, screenshot_path: str, error_log: str) -> Dict:
        # Convert image to base64 for multimodal input
        with open(screenshot_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        # Use vision model to understand the UI context
        vision_prompt = f"Analyze this UI screenshot and error log: {error_log}"
        vision_analysis = ollama.generate(
            model=self.models['vision'],
            prompt=vision_prompt,
            images=[image_data]
        )
        
        # Pass analysis to coding specialist for fixes
        code_prompt = f"""
        Based on this UI analysis: {vision_analysis['response']}
        Provide specific code fixes for common frontend frameworks.
        Focus on React/Vue/ Angular solutions.
        """
        
        return {
            'vision_analysis': vision_analysis['response'],
            'code_fixes': ollama.generate(model=self.models['coding'], prompt=code_prompt)['response']
        }

# Usage example
