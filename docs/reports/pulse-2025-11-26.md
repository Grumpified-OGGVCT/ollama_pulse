import ollama
import base64
from PIL import Image
import io

class MultiModalDeveloper:
    def __init__(self):
        self.vl_model = "qwen3-vl:235b-cloud"
        self.coder_model = "qwen3-coder:480b-cloud"
        self.agent_model = "glm-4.6:cloud"
    
    def analyze_code_with_screenshot(self, code_path, screenshot_path):
        # Convert screenshot to base64 for the vision model
        with Image.open(screenshot_path) as img:
            buffered = io.BytesIO()
            img.save(buffered, format="PNG")
            img_str = base64.b64encode(buffered.getvalue()).decode()
        
        # Get visual analysis
        vision_prompt = f"""
        Analyze this UI screenshot and describe the components and layout.
        Then correlate with this code file: {code_path}
        """
        
        vision_response = ollama.chat(
            model=self.vl_model,
            messages=[{
                "role": "user",
                "content": vision_prompt,
                "images": [img_str]
            }]
        )
        
        # Use coding model for implementation suggestions
        coder_prompt = f"""
        Based on this visual analysis: {vision_response['message']['content']}
        Suggest specific code improvements and optimizations.
        """
        
        return ollama.chat(model=self.coder_model, messages=[{
            "role": "user", 
            "content": coder_prompt
        }])

    def cross_language_refactor(self, source_code, source_lang, target_lang):
        prompt = f"""
        Convert this {source_lang} code to {target_lang}, maintaining functionality
        and following best practices for {target_lang}:
        
        {source_code}
        """
        
        response = ollama.chat(
            model=self.coder_model,
            messages=[{"role": "user", "content": prompt}]
        )
        return response['message']['content']
