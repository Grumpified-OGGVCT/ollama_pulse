                    ]
                }
            ]
        )
        
        return response['message']['content']

class DevelopmentOrchestrator:
    def __init__(self):
        self.models = {
            "manager": "glm-4.6:cloud",
            "coder": "qwen3-coder:480b-cloud", 
            "optimizer": "minimax-m2:cloud"
        }
    
    def coordinate_development_task(self, task_description):
        # Step 1: Planning with manager model
        plan = ollama.chat(
            model=self.models["manager"],
            messages=[{
                "role": "user",
                "text": f"Break this task into subtasks: {task_description}"
            }]
        )
        
        # Step 2: Code implementation with specialist
        implementation = ollama.chat(
            model=self.models["coder"],
            messages=[{
                "role": "user", 
                "text": f"Implement this plan: {plan}"
            }]
        )
        
        # Step 3: Optimization review
        optimized = ollama.chat(
            model=self.models["optimizer"],
            messages=[{
                "role": "user",
                "text": f"Optimize this code: {implementation}"
            }]
        )
        
        return {
            "plan": plan,
            "implementation": implementation,
            "optimized": optimized
        }

# Usage
orchestrator = DevelopmentOrchestrator()
result = orchestrator.coordinate_development_task(
    "Create a React component for a data table with sorting and pagination"
)
