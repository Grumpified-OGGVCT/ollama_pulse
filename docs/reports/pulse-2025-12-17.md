```python
import requests
import base64
from typing import Dict, List

class MultiModalCodeAssistant:
    def __init__(self, ollama_base_url: str = "http://localhost:11434"):
        self.base_url = ollama_base_url
        self.vision_model = "qwen3-vl:235b-cloud"
        self.code_model = "qwen3-coder:480b-cloud"
    
    def analyze_screenshot_and_code(self, screenshot_path: str, code_snippet: str) -> Dict:
        # Convert screenshot to base64
        with open(screenshot_path, "rb") as img_file:
            img_base64 = base64.b64encode(img_file.read()).decode('utf-8')
        
        # Vision analysis
        vision_prompt = f"""
        Analyze this UI screenshot and identify:
        1. Visual inconsistencies or bugs
        2. UI/UX improvements
        3. Accessibility issues
        
        Base64 image: {img_base64}
        """
        
        vision_response = self._call_model(self.vision_model, vision_prompt)
        
        # Code analysis with context from vision
        code_prompt = f"""
        Given this code:
        {code_snippet}
        
        And these UI issues identified:
        {vision_response}
        
        Suggest specific code fixes that address both functional and visual concerns.
        """
        
        code_response = self._call_model(self.code_model, code_prompt)
        
        return {
            "ui_analysis": vision_response,
            "code_fixes": code_response
        }
    
    def _call_model(self, model: str, prompt: str) -> str:
        response = requests.post(
            f"{self.base_url}/api/generate",
            json={"model": model, "prompt": prompt, "stream": False}
        )
        return response.json()["response"]

# Usage example
assistant = MultiModalCodeAssistant()
result = assistant.analyze_screenshot_and_code(

)
```

```python
import os
from pathlib import Path

class CodebaseAnalyzer:
    def __init__(self):
        self.model = "glm-4.6:cloud"
        self.max_context = 200000  # 200K tokens
    
    def analyze_directory(self, directory_path: str, focus: str = "architecture") -> str:
        """Analyze entire codebase within context limits"""
        code_context = self._build_code_context(directory_path)
        
        prompt = f"""
        Analyze this codebase for {focus}:
        
        {code_context}
        
        Provide insights on:
        1. Architectural patterns and anti-patterns
        2. Potential performance bottlenecks  
        3. Security concerns
        4. Code quality recommendations
        """
        
        return self._call_model(prompt)
    
    def _build_code_context(self, directory_path: str) -> str:
        context = ""
        for file_path in Path(directory_path).rglob("*.py"):
            if len(context) < self.max_context * 0.8:  # Leave room for response
                with open(file_path, 'r') as f:
                    context += f"File: {file_path}\n```python\n{f.read()}\n```\n\n"
        return context
```
