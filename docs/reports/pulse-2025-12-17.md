### Context Window Management for Large Codebases

```python
def chunk_codebase(codebase: str, max_tokens: int = 120000) -> List[str]:
    """Split large codebases to fit within model context limits"""
    # Simple chunking - you'd want more sophisticated AST-based splitting
    lines = codebase.split('\n')
    chunks = []
    current_chunk = []
    current_length = 0
    
    for line in lines:
        if current_length + len(line) < max_tokens:
            current_chunk.append(line)
            current_length += len(line)
        else:
            chunks.append('\n'.join(current_chunk))
            current_chunk = [line]
            current_length = len(line)
    
    if current_chunk:
        chunks.append('\n'.join(current_chunk))
    
    return chunks

async def analyze_large_codebase(codebase: str, task: str):
    """Use glm-4.6's 200K context for large-scale code analysis"""
    chunks = chunk_codebase(codebase)
    analyses = []
    
    for chunk in chunks:
        analysis = await orchestrator.route_task(
            f"{task}. Analyze this code section:",
            chunk
        )
        analyses.append(analysis)
    
    # Combine analyses using the reasoning model
    combined_analysis = await orchestrator.route_task(
        "Synthesize these code analyses into a comprehensive report",
        "\n\n".join(analyses)
    )
    
    return combined_analysis
```
