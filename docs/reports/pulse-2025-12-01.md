import ollama
import asyncio
from typing import List, Dict

class MultiModelDeveloper:
    def __init__(self):
        self.models = {
            'vision': 'qwen3-vl:235b-cloud',
            'reasoning': 'glm-4.6:cloud', 
            'coding': 'qwen3-coder:480b-cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def analyze_architectural_decision(self, code_snippet: str, diagram_path: str) -> Dict:
        """Use multiple models to analyze code against architecture diagrams"""
        
        # Step 1: Visual analysis
        vision_prompt = f"""
        Analyze this architecture diagram at {diagram_path}. 
        Describe the components and their relationships.
        """
        visual_context = await ollama.generate(
            model=self.models['vision'],
            prompt=vision_prompt
        )
        
        # Step 2: Reasoning about alignment
        reasoning_prompt = f"""
        Code snippet: {code_snippet}
        Architecture context: {visual_context}
        
        Analyze if the code aligns with the intended architecture.
        Identify any violations or improvements needed.
        """
        alignment_analysis = await ollama.generate(
            model=self.models['reasoning'],
            prompt=reasoning_prompt
        )
        
        # Step 3: Generate fixes if needed
        if "violation" in alignment_analysis.lower() or "improvement" in alignment_analysis.lower():
            fix_prompt = f"""
            Based on this analysis: {alignment_analysis}
            Suggest specific code changes to better align with the architecture.
            Provide the updated code.
            """
            fixes = await ollama.generate(
                model=self.models['coding'],
                prompt=fix_prompt
            )
            
            return {
                'alignment_issues': True,
                'analysis': alignment_analysis,
                'suggested_fixes': fixes
            }
        
        return {'alignment_issues': False, 'analysis': alignment_analysis}

# Usage example
async def main():
    developer = MultiModelDeveloper()
    
    result = await developer.analyze_architectural_decision(
        code_snippet="public class UserService { ... }",
        diagram_path="/architecture/microservices.png"
    )
    
    print(f"Analysis complete: {result}")

if __name__ == "__main__":
    asyncio.run(main())
