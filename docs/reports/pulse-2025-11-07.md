        }
    
    async def analyze_technical_diagram(self, image_path: str, requirements: str):
        """Use vision model to understand diagrams and generate code"""
        
        # Step 1: Vision analysis
        vision_prompt = f"""
        Analyze this technical architecture diagram and describe:
        1. Components and their relationships
        2. Data flow patterns
        3. Potential implementation technologies
        
        Requirements: {requirements}
        """
        
        vision_response = await ollama.generate(
            model=self.models['vision'],
            prompt=vision_prompt,
            images=[image_path]
        )
        
        # Step 2: Code generation based on analysis
        code_prompt = f"""
        Based on this architecture analysis: {vision_response}
        
        Generate implementation code with:
        - Main service structure
        - API endpoints
        - Database schema
        
        Use modern best practices.
        """
        
        code_response = await ollama.generate(
            model=self.models['coding'],
            prompt=code_prompt
        )
        
        return {
            'analysis': vision_response,
            'implementation': code_response
        }
