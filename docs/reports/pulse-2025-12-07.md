```python
import ollama
import asyncio
from typing import List, Dict

class MultiModelDevelopmentAssistant:
    def __init__(self):
        self.models = {
            'planning': 'glm-4.6:cloud',
            'coding': 'qwen3-coder:480b-cloud', 
            'review': 'gpt-oss:20b-cloud',
            'visual': 'qwen3-vl:235b-cloud'
        }
    
    async def plan_and_implement(self, feature_description: str, existing_code: str = ""):
        """Chain models to plan and implement a feature"""
        
        # Step 1: Planning with GLM-4.6 (agentic reasoning)
        planning_prompt = f"""
        As technical lead, break down this feature into concrete tasks:
        Feature: {feature_description}
        
        Existing code context: {existing_code[:5000]}...
        
        Provide a structured plan with:
        1. Architecture decisions needed
        2. Specific files/functions to modify
        3. Potential risks and edge cases
        """
        
        plan = await self.call_model('planning', planning_prompt)
        
        # Step 2: Implementation with Qwen3-Coder
        implementation_prompt = f"""
        Based on this plan, implement the core functionality:
        
        Plan: {plan}
        
        Write production-ready code with:
        - Proper error handling
        - Type annotations (if applicable)
        - Clear documentation
        - Following existing patterns from: {existing_code[:2000]}...
        """
        
        implementation = await self.call_model('coding', implementation_prompt)
        
        # Step 3: Review with GPT-OSS
        review_prompt = f"""
        Review this implementation for quality and consistency:
        
        Original Plan: {plan}
        Implementation: {implementation}
        
        Check for:
        - Code quality issues
        - Security concerns  
        - Performance optimizations
        - Consistency with existing patterns
        """
        
        review = await self.call_model('review', review_prompt)
        
        return {
            'plan': plan,
            'implementation': implementation,
            'review': review
        }
    
    async def call_model(self, model_type: str, prompt: str) -> str:
        response = ollama.chat(
            model=self.models[model_type],
            messages=[{'role': 'user', 'content': prompt}]
        )
        return response['message']['content']

# Usage example
async def main():
    assistant = MultiModelDevelopmentAssistant()
    
    result = await assistant.plan_and_implement(
        "Add user authentication with JWT tokens to our Flask API",
        existing_code="...your existing codebase snippet..."
    )
    
    print("Plan:", result['plan'])
    print("\nImplementation:", result['implementation']) 
    print("\nReview:", result['review'])
```
