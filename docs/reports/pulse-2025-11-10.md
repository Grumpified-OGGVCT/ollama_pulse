```python
            'coding': 'qwen3-coder:480b-cloud',
            'reasoning': 'glm-4.6:cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    def analyze_code_with_context(self, code_file_path, question):
        """Use qwen3-coder's massive context for deep code analysis"""
        with open(code_file_path, 'r') as f:
            code_content = f.read()
        
        prompt = f"""
        Analyze this code and answer: {question}
        
        Code:
        {code_content}
        """
        
        response = ollama.chat(
            model=self.models['coding'],
            messages=[{'role': 'user', 'content': prompt}]
        )
        return response['message']['content']
    
    def visual_code_review(self, screenshot_path, code_snippet):
        """Combine vision and code analysis for comprehensive reviews"""
        # Convert image to base64
        with open(screenshot_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        prompt = f"""
        Analyze this UI screenshot alongside the code that generates it.
        
        Code:
        {code_snippet}
        
        [IMAGE]{image_data}[/IMAGE]
        
        Identify any visual inconsistencies, accessibility issues, or potential improvements.
        """
        
        response = ollama.chat(
            model=self.models['vision'],
            messages=[{'role': 'user', 'content': prompt}]
        )
        return response['message']['content']
```

```python
def complex_problem_solver(problem_description, visual_assets=None):
    """Chain multiple models for complex problem solving"""
    
    # Step 1: Use reasoning model to break down the problem
    reasoning_prompt = f"Break this problem into steps: {problem_description}"
    plan = ollama.chat(model='glm-4.6:cloud', messages=[
        {'role': 'user', 'content': reasoning_prompt}
    ])
    
    # Step 2: If visual elements, analyze with vision model
    if visual_assets:
        vision_analysis = ollama.chat(model='qwen3-vl:235b-cloud', messages=[
            {'role': 'user', 'content': f"Analyze these assets: {visual_assets}"}
        ])
    
    # Step 3: Generate code solution with coding specialist
    code_prompt = f"Based on plan: {plan} and visual analysis: {vision_analysis if visual_assets else 'None'}, write the solution code"
    solution = ollama.chat(model='qwen3-coder:480b-cloud', messages=[
        {'role': 'user', 'content': code_prompt}
    ])
    
    return solution
```

**Pain Point #1: Context Window Limitations**
Remember trying to analyze large codebases and hitting token limits? qwen3-coder's 262K context means you can process entire medium-sized projects in one go. No more chunking, no more losing context between sections.

**Pain Point #2: Siloed Model Capabilities**
Previously, you'd need separate models for vision, coding, and reasoning. Today's hybrids mean one model can understand your screenshot, analyze the related code, and reason about the fix - maintaining context throughout.

**Pain Point #3: Specialized vs General Trade-offs**
The gpt-oss model specifically addresses the "jack of all trades, master of none" problem. At 20B parameters, it's optimized for developer use cases without being overly specialized.

**Pain Point #4: Agent Workflow Efficiency**
minimax-m2 and glm-4.6 directly tackle the inefficiency of multi-step agentic workflows. Instead of costly back-and-forth between specialized models, these can handle complex reasoning chains more efficiently.
