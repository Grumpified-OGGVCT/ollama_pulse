    def process_technical_document(self, image_path: str, query: str) -> str:
        # Step 1: Visual analysis
        vision_prompt = f"""Analyze this technical diagram and extract:
        - Components and their relationships
        - Data flow patterns
        - Key architectural decisions"""
        
        visual_analysis = ollama.generate(
            model=self.models['vision'],
            prompt=vision_prompt,
            images=[image_path]
        )
        
        # Step 2: Reasoning about implications
        reasoning_prompt = f"""Based on this analysis: {visual_analysis['response']}
        Answer this query: {query}
        Consider scalability, security, and maintenance implications."""
        
        reasoning = ollama.generate(
            model=self.models['reasoning'],
            prompt=reasoning_prompt
        )
        
        # Step 3: Generate implementation if needed
        if "implement" in query.lower():
            code_prompt = f"""Create production-ready code based on: {reasoning['response']}
            Focus on clean architecture and best practices."""
            
            implementation = ollama.generate(
                model=self.models['coding'],
                prompt=code_prompt
            )
            return implementation['response']
        
        return reasoning['response']

**Integration Pattern: Model Routing Based on Content Type**
```python
def smart_model_router(content: str, content_type: str) -> str:
    """Route requests to optimal model based on content characteristics"""
    
    if content_type == "image" or "diagram" in content.lower():
        return self.models['vision']
    
    elif len(content) > 50000:  # Large codebases
        return self.models['coding']
    
    elif "reasoning" in content.lower() or "analysis" in content.lower():
        return self.models['reasoning']
    
    else:
        return self.models['general']
```
