```python
import ollama
import base64
from typing import List

class MultiModalAgent:
    def __init__(self):
        self.vision_model = "qwen3-vl:235b-cloud"
        self.reasoning_model = "glm-4.6:cloud"
        self.coding_model = "qwen3-coder:480b-cloud"
    
    def process_document_system(self, image_path: str, query: str) -> str:
        # Encode image for multimodal processing
        with open(image_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        # First, extract visual information
        vision_response = ollama.chat(
            model=self.vision_model,
            messages=[{
                "role": "user",
                "content": [
                    {"type": "text", "text": f"Analyze this document and extract key information: {query}"},
                    {"type": "image", "source": f"data:image/jpeg;base64,{image_data}"}
                ]
            }]
        )
        
        # Then, use reasoning model for decision making
        reasoning_response = ollama.chat(
            model=self.reasoning_model,
            messages=[{
                "role": "user",
                "content": f"Based on this document analysis: {vision_response['message']['content']}. Make a decision and explain your reasoning."
            }]
        )
        
        return reasoning_response['message']['content']

# Usage example
agent = MultiModalAgent()
result = agent.process_document_system("invoice.jpg", "Extract vendor, amount, and due date")
print(result)
```
