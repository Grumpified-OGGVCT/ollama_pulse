```python
import ollama
import asyncio
from typing import List, Dict

class MultiModelDeveloper:
    def __init__(self):
        self.models = {
            'rapid_prototype': 'minimax-m2:cloud',
            'code_review': 'gpt-oss:20b-cloud', 
            'optimization': 'qwen3-coder:480b-cloud',
            'planning': 'glm-4.6:cloud'
        }
    
    async def develop_feature(self, feature_description: str):
        # Step 1: Planning with agentic reasoning
        plan_prompt = f"""Break down this feature into implementable tasks: {feature_description}
        Consider file structure, dependencies, and testing strategy."""
        
        plan = await ollama.generate(
            model=self.models['planning'],
            prompt=plan_prompt
        )
        
        # Step 2: Rapid prototyping
        prototype_prompt = f"""Based on this plan, create initial implementation:
        Plan: {plan}
        Feature: {feature_description}
        Focus on working code over perfection."""
        
        prototype = await ollama.generate(
            model=self.models['rapid_prototype'],
            prompt=prototype_prompt
        )
        
        # Step 3: Code review and refinement
        review_prompt = f"""Review this code for best practices and potential issues:
        Code: {prototype}
        Provide specific improvement suggestions."""
        
        improvements = await ollama.generate(
            model=self.models['code_review'],
            prompt=review_prompt
        )
        
        # Step 4: Final optimization
        final_prompt = f"""Optimize this code incorporating feedback:
        Original: {prototype}
        Improvements: {improvements}
        Focus on performance, readability, and scalability."""
        
        final_code = await ollama.generate(
            model=self.models['optimization'],
            prompt=final_prompt
        )
        
        return {
            'plan': plan,
            'prototype': prototype,
            'review': improvements,
            'final_code': final_code
        }

# Usage example
developer = MultiModelDeveloper()
result = asyncio.run(developer.develop_feature(
    "Create a REST API endpoint for user authentication with JWT tokens"
))
```
