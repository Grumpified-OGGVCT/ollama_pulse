```python
import ollama
import asyncio
from typing import List, Dict

class ModelOrchestrator:
    def __init__(self):
        self.specialists = {
            'vision': 'qwen3-vl:235b-cloud',
            'coding': 'qwen3-coder:480b-cloud', 
            'planning': 'glm-4.6:cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def consult_specialist(self, specialist: str, prompt: str) -> str:
        """Consult a specialized model for its domain expertise"""
        response = ollama.chat(
            model=self.specialists[specialist],
            messages=[{'role': 'user', 'content': prompt}]
        )
        return response['message']['content']
    
    async def build_feature(self, feature_description: str) -> Dict:
        """Orchestrate multiple models to build a feature"""
        # Planning phase
        plan = await self.consult_specialist(
            'planning', 
            f"Break this feature into implementable tasks: {feature_description}"
        )
        
        # Coding phase
        implementation = await self.consult_specialist(
            'coding',
            f"Implement this plan: {plan}"
        )
        
        # Review phase
        review = await self.consult_specialist(
            'general',
            f"Review this code for quality and security: {implementation}"
        )
        
        return {
            'plan': plan,
            'implementation': implementation,
            'review': review
        }

# Usage example
orchestrator = ModelOrchestrator()
result = asyncio.run(orchestrator.build_feature(
    "Create a REST API for user management with authentication"
))
```

    """Chain multiple specialized models for complex tasks"""

    for model in models:
        prompt = f"Based on this context: {context}. Provide your specialized contribution."
        response = ollama.chat(model=model, messages=[{'role': 'user', 'content': prompt}])
