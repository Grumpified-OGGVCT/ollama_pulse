                "role": "user",

            }]
        )
        return response['message']['content']

    def generate_docs_from_screenshot(self, screenshot_path, code_context):

        with open(screenshot_path, "rb") as img_file:

        prompt = {
            "role": "user",
            "content": [
                {"type": "text", "text": f"Code context: {code_context}\n\nDescribe what this UI does and generate usage documentation:"},
                {"type": "image", "source": {"type": "base64", "data": img_base64}}
            ]
        }
        
        response = ollama.chat(model=self.vision, messages=[prompt])
        return response['message']['content']
