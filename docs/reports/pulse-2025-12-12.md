### Basic Multi-Model Orchestration Pattern

```python
import ollama
import json
from typing import Dict, Any

class DevelopmentOrchestrator:
    def __init__(self):
        self.models = {
            'vision_analysis': 'qwen3-vl:235b-cloud',
            'complex_reasoning': 'glm-4.6:cloud',
            'code_generation': 'qwen3-coder:480b-cloud',
            'rapid_prototyping': 'gpt-oss:20b-cloud',
            'efficient_coding': 'minimax-m2:cloud'
        }
    
    def route_task(self, task_description: str, context: Dict[str, Any]) -> str:
        """Intelligently route tasks to the best model"""
        
        # Simple routing logic - expand this based on your needs
        if any(keyword in task_description.lower() for keyword in ['image', 'visual', 'screenshot', 'diagram']):
            return self.models['vision_analysis']
        elif any(keyword in task_description.lower() for keyword in ['architecture', 'planning', 'reasoning']):
            return self.models['complex_reasoning']
        elif 'refactor' in task_description.lower() or 'optimize' in task_description.lower():
            return self.models['efficient_coding']
        else:
            return self.models['code_generation']
    
    def execute_task(self, task: str, context: str = "") -> str:
        best_model = self.route_task(task, {})
        
        prompt = f"""
        Context: {context}
        Task: {task}
        
        Please provide a complete, working solution.
        """
        
        response = ollama.generate(model=best_model, prompt=prompt)
        return response['response']

# Usage example
```
