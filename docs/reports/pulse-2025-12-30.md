    def analyze_ui_and_generate_code(self, screenshot_path: str, requirements: str) -> str:
        # Convert image to base64 for multimodal input
        with open(screenshot_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        # Use VL model to understand the visual requirements

        Analyze this UI screenshot and describe the components, layout, and functionality needed.

                "content": [

                ]
            }]
        )
        
        # Pass analysis to coding specialist
        code_prompt = f"""
        Based on this UI analysis, generate complete React TypeScript components with Tailwind CSS.
        
        Analysis: {vision_analysis['message']['content']}
        
        Requirements: {requirements}
        """
        
        return ollama.chat(
            model=self.coder_model,
            messages=[{"role": "user", "content": code_prompt}]
        )

def codebase_analysis_agent(codebase_path: str, analysis_goal: str) -> Dict:
    """Use GLM-4.6's 200K context to analyze entire codebases"""
    
    # Read and chunk large codebases intelligently
    code_files = []
    for root, dirs, files in os.walk(codebase_path):
        for file in files:
            if file.endswith(('.py', '.js', '.ts', '.java')):
                filepath = os.path.join(root, file)
                with open(filepath, 'r', encoding='utf-8') as f:
                    content = f.read()
                    code_files.append(f"File: {filepath}\nContent:\n{content}")
    
    # Smart chunking for 200K context
    chunks = []
    current_chunk = ""
    for file_content in code_files:
        if len(current_chunk + file_content) < 180000:  # Leave room for prompts
            current_chunk += file_content + "\n\n"
        else:
            chunks.append(current_chunk)
            current_chunk = file_content + "\n\n"
    
    analyses = []
    for chunk in chunks:
        analysis = ollama.chat(
            model="glm-4.6:cloud",
            messages=[{
                "role": "user", 
                "content": f"""
                Analyze this codebase chunk for: {analysis_goal}
                
                Provide specific recommendations for:
                - Architecture improvements
                - Security vulnerabilities  
                - Performance optimizations
                - Code quality issues
                
                Codebase excerpt:
                {chunk}
                """
            }]
        )
        analyses.append(analysis['message']['content'])
    
    return {"chunk_analyses": analyses, "total_files_processed": len(code_files)}
