```python
import requests
import json
from typing import Dict, List

class OllamaOrchestrator:
    def __init__(self, base_url: str = "http://localhost:11434"):
        self.base_url = base_url
        self.models = {
            "vision": "qwen3-vl:235b-cloud",
            "coding": "qwen3-coder:480b-cloud", 
            "agentic": "glm-4.6:cloud",
            "general": "gpt-oss:20b-cloud"
        }
    
    def generate_with_model(self, model: str, prompt: str, images: List[str] = None):
        payload = {
            "model": self.models[model],
            "prompt": prompt,
            "stream": False
        }
        
        if images and model == "vision":
            payload["images"] = images
            
        response = requests.post(f"{self.base_url}/api/generate", json=payload)
        return response.json()["response"]

# Usage example: Multi-step code generation with visual input

    images=["screenshot.png"]
)

# Step 2: Generate the fix using the coding specialist

    "coding",
    f"Based on this UI issue: {ui_analysis}, generate the React component fix"
)
```
