import base64
from typing import List, Dict

class MultiModalDeveloper:
    def __init__(self):
        self.vision_model = "qwen3-vl:235b-cloud"
        self.agent_model = "glm-4.6:cloud"
        self.coder_model = "qwen3-coder:480b-cloud"
    
    def analyze_ui_issue(self, screenshot_path: str, code_context: str) -> Dict:
        """Analyze a UI screenshot with code context"""
        with open(screenshot_path, "rb") as f:
            image_data = base64.b64encode(f.read()).decode()
        
        vision_prompt = f"""
        Analyze this UI screenshot and identify:
        1. Visual inconsistencies or bugs
        2. UI components visible
        3. Layout issues
        
        Code context: {code_context}
        """
        
        vision_response = ollama.generate(
            model=self.vision_model,
            prompt=vision_prompt,
            images=[image_data]
        )
        
        # Pass to agent for reasoning
        agent_prompt = f"""
        Based on this UI analysis, suggest specific code fixes:
        
        UI Analysis: {vision_response['response']}
        Code Context: {code_context}
        
        Provide concrete file changes with line numbers.
        """
        
        agent_response = ollama.generate(
            model=self.agent_model,
            prompt=agent_prompt
        )
        
        return {
            'visual_analysis': vision_response['response'],
            'code_fixes': agent_response['response']
        }

# Usage example
