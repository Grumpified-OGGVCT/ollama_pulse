import asyncio
from typing import List, Dict

class DevelopmentOrchestrator:
    def __init__(self):
        self.models = {
            'architect': 'glm-4.6:cloud',      # Reasoning & planning
            'coder': 'qwen3-coder:480b-cloud', # Implementation
            'reviewer': 'gpt-oss:20b-cloud',   # Quality assurance
            'vision': 'qwen3-vl:235b-cloud'    # Multimodal analysis
        }
    
    async def solve_problem(self, problem_description: str, screenshot_path: str = None):
        """Orchestrate multiple models to solve a development problem"""
        
        tasks = []
        
        if screenshot_path:
            # Use vision model to analyze visual context
            vision_task = self.analyze_visual_context(screenshot_path, problem_description)
            tasks.append(vision_task)
        
        # Architect plans the solution
        planning_task = self.plan_solution(problem_description)
        tasks.append(planning_task)
        
        results = await asyncio.gather(*tasks)
        
        # Combine insights and generate code
        combined_context = "\n".join(results)
        implementation = await self.generate_implementation(combined_context)
        
        return await self.review_solution(implementation, problem_description)
    
    async def analyze_visual_context(self, image_path: str, problem: str):
        response = ollama.chat(
            model=self.models['vision'],
            messages=[{
                'role': 'user',
                'content': f"""
                Analyze this screenshot in the context of this problem: {problem}
                Describe any visual elements that might be relevant to the solution.
                """,
                'images': [image_path]
            }]
        )
        return f"Visual Analysis: {response['message']['content']}"
    
    async def plan_solution(self, problem: str):
        response = ollama.chat(
            model=self.models['architect'],
            messages=[{
                'role': 'user',
                'content': f"""
                Break down this problem into a step-by-step implementation plan: {problem}
                Consider architecture, dependencies, and potential pitfalls.
                """
            }]
        )
        return f"Architecture Plan: {response['message']['content']}"

# Usage example
async def main():
    orchestrator = DevelopmentOrchestrator()
    
    problem = "Create a REST API for user management with JWT authentication"
    solution = await orchestrator.solve_problem(problem)
    
    print("Generated Solution:")
    print(solution)

# Run it
if __name__ == "__main__":
    asyncio.run(main())
