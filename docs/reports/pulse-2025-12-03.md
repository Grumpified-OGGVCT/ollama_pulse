```python
import asyncio
from typing import Dict, Any

class MultiModelOrchestrator:
    def __init__(self):
        self.models = {
            'vision': 'qwen3-vl:235b-cloud',
            'reasoning': 'glm-4.6:cloud', 
            'coding': 'qwen3-coder:480b-cloud',
            'general': 'gpt-oss:20b-cloud',
            'efficient': 'minimax-m2:cloud'
        }
    
    async def analyze_technical_diagram(self, image_path: str, question: str):
        """Use vision model to understand diagrams, then coding model for implementation"""
        
        # Step 1: Vision analysis
        vision_prompt = f"Describe this technical diagram and identify key components: {question}"
        vision_response = await ollama.generate(
            model=self.models['vision'],
            prompt=vision_prompt,
            images=[image_path]
        )
        
        # Step 2: Code generation based on vision analysis
        coding_prompt = f"""
        Based on this architecture analysis: {vision_response}
        Generate Python implementation code for the core components.
        Focus on clean, maintainable code with proper error handling.
        """
        
        code_response = await ollama.generate(
            model=self.models['coding'],
            prompt=coding_prompt
        )
        
        return {
            'architecture_analysis': vision_response,
            'implementation_code': code_response
        }

    async def refactor_agent(self, code_snippet: str, requirements: str):
        """Use reasoning model to plan refactoring, efficient model to execute"""
        
        # Planning phase with reasoning model
        plan_prompt = f"""
        Code to refactor: {code_snippet}
        Requirements: {requirements}
        
        Create a step-by-step refactoring plan considering:
        1. Performance optimization
        2. Code readability
        3. Maintainability
        4. Testing strategy
        """
        
        refactor_plan = await ollama.generate(
            model=self.models['reasoning'],
            prompt=plan_prompt
        )
        
        # Execution phase with efficient model
        execution_prompt = f"""
        Refactoring plan: {refactor_plan}
        Original code: {code_snippet}
        
        Implement the refactored code following the plan.
        """
        
        refactored_code = await ollama.generate(
            model=self.models['efficient'],
            prompt=execution_prompt
        )
        
        return {
            'refactoring_plan': refactor_plan,
            'refactored_code': refactored_code
        }

# Usage example
async def main():
    orchestrator = MultiModelOrchestrator()
    
    # Analyze a system diagram and get code
    result = await orchestrator.analyze_technical_diagram(
        image_path="architecture.png",
        question="How would we implement the API gateway shown here?"
    )
    
    print(f"Analysis: {result['architecture_analysis']}")
    print(f"Generated code: {result['implementation_code']}")

if __name__ == "__main__":
    asyncio.run(main())
```
