import ollama
import asyncio
from typing import List, Dict

class MultiModelCodingAssistant:
    def __init__(self):
        self.models = {
            'code_review': 'qwen3-coder:480b-cloud',
            'architecture': 'glm-4.6:cloud', 
            'visual_analysis': 'qwen3-vl:235b-cloud',
            'general_dev': 'gpt-oss:20b-cloud'
        }
    
    async def analyze_codebase(self, file_paths: List[str], task_description: str):
        """Use specialized models for different aspects of code analysis"""
        
        # Read all files into context
        context = ""
        for file_path in file_paths:
            with open(file_path, 'r') as f:
                context += f"\n--- {file_path} ---\n{f.read()}"
        
        # Use qwen3-coder for deep code analysis
        code_analysis = await ollama.generate(
            model=self.models['code_review'],
            prompt=f"Analyze this codebase for {task_description}:\n{context}"
        )
        
        # Use glm-4.6 for architectural recommendations
        architecture_review = await ollama.generate(
            model=self.models['architecture'],
            prompt=f"Based on this analysis: {code_analysis}\nSuggest architectural improvements."
        )
        
        return {
            'code_analysis': code_analysis,
            'architecture_review': architecture_review
        }

# Example usage
async def main():
    assistant = MultiModelCodingAssistant()
    
    # Analyze a project for performance improvements
    results = await assistant.analyze_codebase(
        file_paths=['app/main.py', 'app/utils.py', 'app/models.py'],
        task_description="performance bottlenecks and optimization opportunities"
    )
    
    print("Code Analysis:", results['code_analysis'])
    print("Architecture Suggestions:", results['architecture_review'])

# Run it
asyncio.run(main())
