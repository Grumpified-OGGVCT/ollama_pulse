class OllamaOrchestrator:
    def __init__(self, base_url: str = "http://localhost:11434"):
        self.base_url = base_url
        self.models = {
            "vision": "qwen3-vl:235b-cloud",
            "reasoning": "glm-4.6:cloud", 
            "coding": "qwen3-coder:480b-cloud",
            "general": "gpt-oss:20b-cloud",
            "efficient": "minimax-m2:cloud"
        }
    
    def route_task(self, prompt: str, context: Dict[str, Any] = None) -> str:
        """Intelligently route tasks to the best model"""
        prompt_lower = prompt.lower()
        
        if any(keyword in prompt_lower for keyword in ['image', 'screenshot', 'diagram', 'visual']):
            return self.models['vision']
        elif any(keyword in prompt_lower for keyword in ['reason', 'analyze', 'strategize', 'plan']):
            return self.models['reasoning']
        elif any(keyword in prompt_lower for keyword in ['code', 'program', 'function', 'algorithm']):
            return self.models['coding']
        elif len(prompt) > 1000:  # Long context tasks
            return self.models['efficient']
        else:
            return self.models['general']
    
    def generate(self, prompt: str, images: list = None) -> str:
        model = self.route_task(prompt)
        
        payload = {
            "model": model,
            "prompt": prompt,
            "stream": False
        }
        
        if images and model == self.models['vision']:
            payload["images"] = images
        
        response = requests.post(f"{self.base_url}/api/generate", json=payload)
        return response.json()["response"]
