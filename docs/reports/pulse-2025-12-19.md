import ollama
import asyncio
from typing import Dict, List

class MultiModelCoder:
    def __init__(self):
        self.models = {
            'general': 'gpt-oss:20b-cloud',
            'specialized': 'qwen3-coder:480b-cloud', 
            'agentic': 'glm-4.6:cloud',
            'efficient': 'minimax-m2:cloud'
        }
    
    async def route_task(self, task: str, context: str) -> Dict:
        """Route coding tasks to appropriate models"""
        if len(context) > 100000:  # Large codebase analysis
            model = self.models['specialized']
        elif 'refactor' in task.lower() or 'optimize' in task.lower():
            model = self.models['efficient']
        elif 'plan' in task.lower() or 'architecture' in task.lower():
            model = self.models['agentic']
        else:
            model = self.models['general']
        
        response = await ollama.generate(
            model=model,
            prompt=f"Task: {task}\n\nContext: {context[:50000]}",  # Safe context limit
            options={'temperature': 0.1}
        )
        
        return {
            'model_used': model,
            'response': response['response'],
            'confidence': self._calculate_confidence(task, response)
        }
    
    def _calculate_confidence(self, task: str, response: dict) -> float:
        # Simple confidence scoring based on response characteristics
        response_text = response['response']
        if 'sorry' in response_text.lower() or 'cannot' in response_text.lower():
            return 0.3
        elif 'suggest' in response_text.lower() or 'consider' in response_text.lower():
            return 0.7
        else:
            return 0.9
