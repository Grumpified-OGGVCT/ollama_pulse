### Basic Multi-Model Orchestration Pattern

```python
import ollama
import asyncio
from typing import List, Dict

class ModelOrchestrator:
    def __init__(self):
        self.models = {
            'vision': 'qwen3-vl:235b-cloud',
            'coding': 'qwen3-coder:480b-cloud',
            'reasoning': 'glm-4.6:cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def route_task(self, task: str, context: Dict) -> str:
        if any(keyword in task.lower() for keyword in ['image', 'screenshot', 'visual']):
            return await self.call_model('vision', task, context)
        elif any(keyword in task.lower() for keyword in ['code', 'program', 'debug']):
            return await self.call_model('coding', task, context)
        elif any(keyword in task.lower() for keyword in ['analyze', 'reason', 'plan']):
            return await self.call_model('reasoning', task, context)
        else:
            return await self.call_model('general', task, context)
    
    async def call_model(self, model_type: str, prompt: str, context: Dict) -> str:
        response = ollama.chat(
            model=self.models[model_type],
            messages=[
                {
                    'role': 'system',
                    'content': f"Context: {context}\n\nYou are a specialist in {model_type} tasks."
                },
                {
                    'role': 'user', 
                    'content': prompt
                }
            ]
        )
        return response['message']['content']

# Usage example
orchestrator = ModelOrchestrator()
result = asyncio.run(orchestrator.route_task(
    "Analyze this code snippet and suggest optimizations: def calculate...",
    {'project_type': 'web_app', 'language': 'python'}
))
```
