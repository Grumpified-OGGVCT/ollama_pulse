## üîß How can we leverage these tools?

Let's get practical with some real integration patterns. Here's a Python example showing how you might orchestrate multiple models:

```python
import ollama
import asyncio
from typing import Dict, Any

class MultiModelOrchestrator:
    def __init__(self):
        self.models = {
            'vision': 'qwen3-vl:235b-cloud',
            'reasoning': 'glm-4.6:cloud', 
            'coding': 'qwen3-coder:480b-cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def analyze_ui_issue(self, screenshot_path: str, code_context: str):
        """Multi-model analysis of UI problems"""
        
        # Step 1: Vision analysis
        vision_prompt = f"""
        Analyze this UI screenshot and describe any visual issues, 
        layout problems, or user experience concerns.
        """
        
        vision_response = await ollama.generate(
            model=self.models['vision'],
            prompt=vision_prompt,
            images=[screenshot_path]
        )
        
        # Step 2: Reasoning about root cause
        reasoning_prompt = f"""
        Vision analysis: {vision_response}
        Code context: {code_context}
        
        Based on the visual issues identified, what are the likely root causes 
        in the code? Consider component hierarchy, state management, and styling issues.
        """
        
        reasoning_response = await ollama.generate(
            model=self.models['reasoning'],
            prompt=reasoning_prompt
        )
        
        # Step 3: Generate fixes
        fix_prompt = f"""
        Issue analysis: {reasoning_response}
        Original code: {code_context}
        
        Provide specific code fixes addressing the identified issues.
        Include both the corrected code and explanation of changes.
        """
        
        code_response = await ollama.generate(
            model=self.models['coding'],
            prompt=fix_prompt
        )
        
        return {
            'vision_analysis': vision_response,
            'root_cause': reasoning_response,
            'fixes': code_response
        }

# Usage example
async def main():
    orchestrator = MultiModelOrchestrator()
    
    result = await orchestrator.analyze_ui_issue(
        screenshot_path='bug_screenshot.png',
        code_context='// React component code here...'
    )
    
    print(f"Identified issue: {result['root_cause']}")
    print(f"Suggested fix: {result['fixes']}")

# Run it
asyncio.run(main())
```
