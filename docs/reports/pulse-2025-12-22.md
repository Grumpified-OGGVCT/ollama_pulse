class OllamaMultiModalClient:
    def __init__(self):
        self.models = {
            'vision': 'qwen3-vl:235b-cloud',
            'coding': 'qwen3-coder:480b-cloud',
            'agentic': 'glm-4.6:cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    def analyze_code_with_context(self, file_paths, question):
        """Use Qwen3-coder's massive context for code analysis"""
        context_content = ""
        for path in file_paths:
            with open(path, 'r') as f:
                context_content += f"\n\n--- {path} ---\n{f.read()}"
        
        prompt = f"""Context from codebase:
{context_content[:250000]}  # Leverage 262K context

Question: {question}"""

        response = ollama.chat(model=self.models['coding'], messages=[
            {'role': 'user', 'content': prompt}
        ])
        return response['message']['content']
    
    def visual_code_review(self, screenshot_path, code_snippet):
        """Combine vision and code analysis"""
        # Convert image to base64 for Qwen3-VL
        with open(screenshot_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        prompt = f"""Analyze this UI screenshot and corresponding code:
        
Code:
{code_snippet}

Identify visual issues and suggest code improvements."""

        response = ollama.chat(model=self.models['vision'], messages=[
            {
                'role': 'user', 
                'content': prompt,
                'images': [image_data]
            }
        ])
        return response['message']['content']

# Usage example
