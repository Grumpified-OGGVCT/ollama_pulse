        """Combine visual understanding with technical reasoning"""

            image_data = base64.b64encode(image_file.read()).decode('utf-8')

            {
                "role": "user", 
                "content": prompt,
                "images": [image_data]
            }
        ])
        return response['message']['content']

class DeveloperWorkflow:

        """Route tasks to the most appropriate model"""
        if "screenshot" in task_description.lower() or "image" in task_description.lower():
            return self.vl_model
        elif "complex algorithm" in task_description or "architecture" in task_description:
            return self.coder_model
