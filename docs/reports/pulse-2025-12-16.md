            'vision': 'qwen3-vl:235b-cloud',
            'coding': 'qwen3-coder:480b-cloud', 
            'planning': 'glm-4.6:cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    def analyze_ui_screenshot(self, image_path, user_question):
        """Use vision model to analyze UI and generate code"""
        with open(image_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        prompt = f"""
        Analyze this UI screenshot and {user_question}.
        Provide React component code that implements this design.
        """
        
        response = ollama.chat(
            model=self.models['vision'],
            messages=[
                {
                    'role': 'user',
                    'content': prompt,
                    'images': [image_data]
                }
            ]
        )
        return response['message']['content']

    def refactor_codebase(self, file_paths, refactor_goal):
        """Use the massive context window for large-scale refactoring"""
        code_context = ""
        for path in file_paths:
            with open(path, 'r') as f:
                code_context += f"\n\n--- {path} ---\n{f.read()}"
        
        # Only send relevant chunks if using local Ollama
        # For cloud models, we can send much more context
        response = ollama.chat(
            model=self.models['coding'],
            messages=[
                {
                    'role': 'user', 
                    'content': f"Refactor goal: {refactor_goal}\n\nCodebase:\n{code_context[:100000]}"
                }
            ]
        )
        return response['message']['content']
