class AIDevelopmentTeam:
    def __init__(self):

            'architect': 'glm-4.6:cloud',

    async def code_review_workflow(self, code_snippet, requirements):
        # Architect analyzes requirements
        arch_prompt = f"""
        Analyze these requirements and create a high-level architecture:
        {requirements}
        
        Code to review: {code_snippet}
        """
        
        architect_response = await ollama.generate(
            model=self.models['architect'],

            options={'num_ctx': 200000}  # Use full context
        )
        
        # Coder suggests implementation improvements
        coder_prompt = f"""
        Based on this architecture: {architect_response}

        """
        
        coder_response = await ollama.generate(
            model=self.models['coder'],
            prompt=coder_prompt,
            options={'num_ctx': 262000}
        )
        
        return {
            'architecture': architect_response,
            'implementation': coder_response
        }
