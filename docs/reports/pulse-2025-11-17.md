class AgenticWorkflow:
    def __init__(self):
        self.model = "glm-4.6:cloud"
        self.conversation_history = []
    
    def execute_workflow(self, task_description):
        """Leverage the agentic capabilities for multi-step tasks"""
        
        system_prompt = """
        You are an autonomous coding assistant. Break down complex tasks 
        into steps and execute them systematically. You have access to 
        file systems, APIs, and other tools.
        """
        
        self.conversation_history.append({"role": "user", "content": task_description})
        
        prompt = {
            "model": self.model,
            "prompt": task_description,
            "context": self.conversation_history[-10:],  # Maintain conversation state
            "system": system_prompt,
            "stream": False
        }
        
        response = requests.post("http://localhost:11434/api/generate", json=prompt)
        return response.json()

# Example: Automated code review and fix workflow
