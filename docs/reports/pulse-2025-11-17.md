            'coding': 'qwen3-coder:480b-cloud',
            'agentic': 'glm-4.6:cloud',
            'general': 'gpt-oss:20b-cloud',
            'efficient': 'minimax-m2:cloud'
        }
    
    async def parallel_analysis(self, prompt: str, model_keys: List[str]) -> Dict[str, str]:
        """Run analysis across multiple models concurrently"""
        async def query_model(model_key: str):
            response = ollama.chat(
                model=self.models[model_key],
                messages=[{'role': 'user', 'content': prompt}]
            )
            return model_key, response['message']['content']
        
        tasks = [query_model(key) for key in model_keys]
        results = await asyncio.gather(*tasks)
        return dict(results)
    
    def build_complex_agent(self, system_prompt: str) -> callable:
        """Create a specialized agent using glm-4.6's advanced reasoning"""
        def agent_executor(user_input: str):
            messages = [
                {'role': 'system', 'content': system_prompt},
                {'role': 'user', 'content': user_input}
            ]
            
            response = ollama.chat(
                model=self.models['agentic'],
                messages=messages,
                options={'num_ctx': 200000}  # Leverage the massive context
            )
            return response['message']['content']
        
        return agent_executor
