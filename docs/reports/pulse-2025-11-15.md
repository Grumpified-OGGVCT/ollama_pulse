```python
    def __init__(self, ollama_base_url: str = "http://localhost:11434"):
        self.base_url = ollama_base_url
        
    def analyze_technical_diagram(self, image_path: str, query: str) -> str:
        """Use qwen3-vl for visual technical analysis"""
        prompt = f"""
        Analyze this technical diagram and {query}. 
        Describe the system architecture, data flow, and potential issues.
        """
        
        response = requests.post(
            f"{self.base_url}/api/generate",
            json={
                "model": "qwen3-vl:235b-cloud",
                "prompt": prompt,
                "images": [image_path],
                "stream": False
            }
        )
        return response.json()["response"]
    
    def generate_implementation(self, requirements: str, context: List[str] = None) -> str:
        """Use qwen3-coder for code generation with extensive context"""
        context_text = "\n".join(context) if context else ""
        
        response = requests.post(
            f"{self.base_url}/api/generate",
            json={
                "model": "qwen3-coder:480b-cloud",
                "prompt": f"Context:\n{context_text}\n\nTask: {requirements}",
                "options": {
                    "num_ctx": 262000  # Leverage massive context window
                },
                "stream": False
            }
        )
        return response.json()["response"]
    
    def create_agentic_workflow(self, goal: str) -> Dict:
        """Use glm-4.6 for creating complex agent workflows"""
        response = requests.post(
            f"{self.base_url}/api/generate",
            json={
                "model": "glm-4.6:cloud",
                "prompt": f"""
                Create a step-by-step agentic workflow to: {goal}
                Break down into discrete, actionable steps with validation checks.
                Return as JSON with steps, dependencies, and success criteria.
                """,
                "stream": False
            }
        )
        return json.loads(response.json()["response"])

# Example usage:
```

```python
def complete_development_cycle(requirement: str, diagram_path: str) -> Dict:
    """Chain multiple models for a comprehensive solution"""
    
    # Step 1: Visual analysis with qwen3-vl
    visual_analysis = assistant.analyze_technical_diagram(diagram_path, requirement)
    
    # Step 2: Agentic planning with glm-4.6
    workflow = assistant.create_agentic_workflow(f"{requirement} based on {visual_analysis}")
    
    # Step 3: Code generation with qwen3-coder
    implementation = assistant.generate_implementation(
        f"Implement {requirement}",
        [visual_analysis, json.dumps(workflow)]
    )
    
    return {
        "analysis": visual_analysis,
        "workflow": workflow,
        "implementation": implementation
    }
```
