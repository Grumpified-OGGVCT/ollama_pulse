```python
import ollama
import asyncio

class MultiModelDevAssistant:
    def __init__(self):
        self.models = {
            'vision': 'qwen3-vl:235b-cloud',
            'reasoning': 'glm-4.6:cloud', 
            'coding': 'qwen3-coder:480b-cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def code_review(self, code: str, screenshots: list = None):
        """Multi-faceted code review using specialized models"""
        tasks = []
        
        if screenshots:
            # Use vision model for UI/UX analysis
            vision_prompt = f"Analyze these UI screenshots with code: {code}"
            tasks.append(self._query_model('vision', vision_prompt, screenshots))
        
        # Use coder model for technical review
        coding_prompt = f"Technical code review for best practices: {code}"
        tasks.append(self._query_model('coding', coding_prompt))
        
        # Use reasoning model for architectural feedback
        reasoning_prompt = f"Architectural implications and scalability: {code}"
        tasks.append(self._query_model('reasoning', reasoning_prompt))
        
        results = await asyncio.gather(*tasks)
        return self._synthesize_feedback(results)
    
    async def _query_model(self, model_type: str, prompt: str, images=None):
        response = ollama.chat(
            model=self.models[model_type],
            messages=[{
                'role': 'user',
                'content': prompt,
                'images': images or []
            }]
        )
        return {
            'model': model_type,
            'response': response['message']['content']
        }

# Usage example

    code="your_python_function_here",
    screenshots=['ui_screenshot.png']
)
```
