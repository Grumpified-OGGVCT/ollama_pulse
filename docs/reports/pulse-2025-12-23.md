import asyncio
from typing import List, Dict, Any
import httpx

class OllamaCloudOrchestrator:
    def __init__(self, base_url: str = "http://localhost:11434"):
        self.base_url = base_url
        self.models = {
            "vision": "qwen3-vl:235b-cloud",
            "reasoning": "glm-4.6:cloud", 
            "coding": "qwen3-coder:480b-cloud",
            "general": "gpt-oss:20b-cloud",
            "efficient": "minimax-m2:cloud"
        }
    
    async def process_multimodal_task(self, image_path: str, code_context: str) -> Dict[str, Any]:
        """Process visual and code inputs together"""
        # First, extract information from the image
        vision_prompt = f"""
        Analyze this technical diagram and describe the system architecture.
        Focus on components, data flow, and potential integration points.
        """
        
        vision_result = await self._call_model(
            self.models["vision"],
            prompt=vision_prompt,
            images=[image_path]
        )
        
        # Then, generate code based on the analysis
        coding_prompt = f"""
        Based on this architecture description: {vision_result}
        And this existing code context: {code_context}
        Generate Terraform configuration for the infrastructure.
        """
        
        code_result = await self._call_model(
            self.models["coding"], 
            prompt=coding_prompt
        )
        
        return {
            "architecture_analysis": vision_result,
            "generated_code": code_result
        }
    
    async def automated_code_review(self, code_snippet: str, test_results: Dict) -> str:
        """Use reasoning model for intelligent code analysis"""
        prompt = f"""
        Analyze this code for quality and issues:
        {code_snippet}
        
        Test results: {test_results}
        
        Consider: performance, security, maintainability, and test coverage.
        Provide specific, actionable recommendations.
        """
        
        return await self._call_model(self.models["reasoning"], prompt)
    
    async def _call_model(self, model: str, prompt: str, images: List[str] = None) -> str:
        """Generic model calling function"""
        async with httpx.AsyncClient() as client:
            data = {
                "model": model,
                "prompt": prompt,
                "stream": False
            }
            
            if images:
                data["images"] = images
                
            response = await client.post(
                f"{self.base_url}/api/generate",
                json=data,
                timeout=30.0
            )
            
            return response.json()["response"]

# Example usage
async def main():
    orchestrator = OllamaCloudOrchestrator()
    
    # Process a system diagram and generate infrastructure code
    result = await orchestrator.process_multimodal_task(
        image_path="architecture.png",
        code_context="Existing AWS Lambda functions"
    )
    
    print(f"Architecture Analysis: {result['architecture_analysis']}")
    print(f"Generated Terraform: {result['generated_code']}")

# Run it
if __name__ == "__main__":
    asyncio.run(main())
