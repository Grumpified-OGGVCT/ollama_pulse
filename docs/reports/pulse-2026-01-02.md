```python
import ollama
import asyncio
from typing import Dict, Any

class MultiModalOrchestrator:
    def __init__(self):
        self.models = {
            'vision': 'qwen3-vl:235b-cloud',
            'coding': 'qwen3-coder:480b-cloud', 
            'reasoning': 'glm-4.6:cloud',
            'efficient': 'minimax-m2:cloud'
        }
    
    async def analyze_codebase(self, file_paths: list[str], screenshots: list[str] = None):
        """Multi-modal code analysis combining visual and textual context"""
        
        # Use vision model to understand UI components
        if screenshots:
            vision_prompt = f"Analyze these UI screenshots and extract component patterns: {screenshots}"
            ui_analysis = await ollama.chat(model=self.models['vision'], messages=[
                {'role': 'user', 'content': vision_prompt}
            ])
        
        # Use coding specialist for deep code analysis
        code_context = "\n".join([open(fp).read() for fp in file_paths])
        code_analysis = await ollama.chat(model=self.models['coding'], messages=[
            {'role': 'user', 'content': f"Analyze this codebase for patterns and improvements:\n{code_context}"}
        ])
        
        # Use reasoning model to synthesize recommendations
        synthesis_prompt = f"""
        Combine UI analysis: {ui_analysis['message']['content']}
        With code analysis: {code_analysis['message']['content']}
        
        Provide specific refactoring recommendations.
        """
        
        return await ollama.chat(model=self.models['reasoning'], messages=[
            {'role': 'user', 'content': synthesis_prompt}
        ])

# Usage example
orchestrator = MultiModalOrchestrator()
result = asyncio.run(orchestrator.analyze_codebase(
    file_paths=['src/components/', 'package.json'],
    screenshots=['design-mockup.png']
))
```
