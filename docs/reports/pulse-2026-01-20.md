import ollama
import asyncio

class MultiModelCoder:
    def __init__(self):
        self.specialists = {
            'refactor': 'glm-4.6:cloud',
            'polyglot': 'qwen3-coder:480b-cloud', 
            'efficiency': 'minimax-m2:cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def get_combined_advice(self, code_input, task_description):
        tasks = []
        for specialist_type, model in self.specialists.items():
            prompt = f"""
            As a {specialist_type} specialist, {task_description}
            
            Code to analyze:
            ```python
            {code_input}
            ```
            
            Provide specific, actionable recommendations.
            """
            
            tasks.append(self._query_model(model, prompt))
        
        results = await asyncio.gather(*tasks)
        return self._synthesize_advice(results)
    
    async def _query_model(self, model, prompt):
        response = ollama.chat(model=model, messages=[{'role': 'user', 'content': prompt}])
        return response['message']['content']
