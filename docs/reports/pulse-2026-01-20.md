    async def analyze_codebase(self, file_paths: List[str], prompt: str):
        """Use multiple models to analyze complex codebases"""
        # First, let the coding specialist understand the structure
        code_context = await self._get_model_analysis(
            self.models["coding"],

        )
        
        # Then have the agentic model create an action plan
        plan = await self._get_model_analysis(
            self.models["agentic"],

        )
        
        return {"analysis": code_context, "plan": plan}
    
    async def _get_model_analysis(self, model: str, prompt: str) -> str:
        async with httpx.AsyncClient() as client:
            response = await client.post(
                f"{self.base_url}/api/generate",
                json={"model": model, "prompt": prompt, "stream": False}
            )
            return response.json()["response"]
