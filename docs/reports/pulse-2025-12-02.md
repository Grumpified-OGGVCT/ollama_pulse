```python
def chunk_context_aware_processing(system_prompt: str, user_input: str, max_chunk_size: int = 50000):
    """Break down large contexts while maintaining coherence"""
    
    # Use glm-4.6's 200K context for large documents
    model = 'glm-4.6:14b-cloud'
    
    # Smart chunking that maintains semantic boundaries
    chunks = []
    current_chunk = ""
    
    # Simple chunking logic (you'd want something smarter in production)
    for paragraph in user_input.split('\n\n'):
        if len(current_chunk) + len(paragraph) < max_chunk_size:
            current_chunk += paragraph + "\n\n"
        else:
            chunks.append(current_chunk)
            current_chunk = paragraph + "\n\n"
    
    if current_chunk:
        chunks.append(current_chunk)
    
    responses = []
    for chunk in chunks:
        response = ollama.chat(model=model, messages=[
            {'role': 'system', 'content': system_prompt},
            {'role': 'user', 'content': chunk}
        ])
        responses.append(response['message']['content'])
    
    return "\n\n".join(responses)
```
