            'vision': 'qwen3-vl:235b-cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def analyze_codebase(self, file_paths: List[str], screenshot_path: str = None):
        """Multi-modal code analysis with visual context"""
        
        # Use vision model to understand UI components
        if screenshot_path:
            vision_prompt = f"Analyze this UI and identify key components: {screenshot_path}"
            ui_analysis = await ollama.generate(
                model=self.models['vision'],
                prompt=vision_prompt
            )
        
        # Use coding specialist for deep code analysis
        code_context = "\n".join([open(fp).read() for fp in file_paths])
        coding_prompt = f"""
        Analyze this codebase with UI context: {ui_analysis if screenshot_path else 'N/A'}
        
        Code:
        {code_context}
        
        Identify architectural issues and suggest improvements.
        """
        
        return await ollama.generate(
            model=self.models['coding'],
            prompt=coding_prompt
        )
    
    async def plan_refactoring(self, analysis_result: str):
        """Use reasoning model to create execution plan"""
        reasoning_prompt = f"""
        Based on this code analysis: {analysis_result}
        
        Create a step-by-step refactoring plan considering:
        1. Risk assessment
        2. Dependency order  
        3. Testing strategy
        4. Rollback options
        """
        
        return await ollama.generate(
            model=self.models['reasoning'],
            prompt=reasoning_prompt
        )

# Usage example
agent = MultiModelDevelopmentAgent()
analysis = await agent.analyze_codebase(
    ['src/main.py', 'src/utils.py'],
    screenshot_path='ui_component.png'
)
