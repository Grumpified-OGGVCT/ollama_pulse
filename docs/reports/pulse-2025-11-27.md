```python
import ollama
import asyncio
from typing import List, Dict

class MultiModelCodeAnalyzer:
    def __init__(self):
        self.models = {
            'code_specialist': 'qwen3-coder:480b-cloud',
            'reasoning_engine': 'glm-4.6:cloud', 
            'vision_analyzer': 'qwen3-vl:235b-cloud'
        }
    
    async def analyze_codebase(self, file_paths: List[str], architecture_diagram: str = None):
        """Multi-stage analysis using specialized models"""
        
        # Stage 1: Vision analysis of architecture diagram
        vision_insights = ""
        if architecture_diagram:
            vision_response = ollama.generate(
                model=self.models['vision_analyzer'],
                prompt=f"Analyze this software architecture diagram: {architecture_diagram}"
            )
            vision_insights = vision_response['response']
        
        # Stage 2: Code analysis with massive context window
        code_context = self._load_files(file_paths)
        code_analysis = ollama.generate(
            model=self.models['code_specialist'],
            prompt=f"""Analyze this codebase with architecture context: {vision_insights}
            
            Code files:
            {code_context}
            
            Identify:
            1. Architectural violations
            2. Performance bottlenecks  
            3. Security concerns
            4. Refactoring opportunities"""
        )
        
        # Stage 3: Agentic reasoning for actionable recommendations
        recommendations = ollama.generate(
            model=self.models['reasoning_engine'],
            prompt=f"""Based on this analysis, create a prioritized action plan:
            
            Analysis: {code_analysis['response']}
            
            Create a sprint plan with:
            - Immediate fixes (week 1)
            - Medium-term improvements (month 1) 
            - Long-term architectural changes (quarter 1)"""
        )
        
        return {
            'vision_analysis': vision_insights,
            'code_analysis': code_analysis['response'],
            'action_plan': recommendations['response']
        }
    
    def _load_files(self, file_paths: List[str]) -> str:
        # Simple file loader - in production, you'd want smarter handling
        content = []
        for path in file_paths:
            try:
                with open(path, 'r') as f:
                    content.append(f"=== {path} ===\n{f.read()}")
            except:
                content.append(f"=== {path} ===\n[Unable to read file]")
        return "\n\n".join(content)

# Usage example
async def main():
    analyzer = MultiModelCodeAnalyzer()
    results = await analyzer.analyze_codebase(
        file_paths=['src/main.py', 'src/utils.py'],
        architecture_diagram='path/to/diagram.png'
    )
    print(results['action_plan'])

# Run it
if __name__ == "__main__":
    asyncio.run(main())
```
