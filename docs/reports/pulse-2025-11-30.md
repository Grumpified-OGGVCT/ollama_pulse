```python
import ollama
import base64
from typing import Dict, Any

class SpecializedModelRouter:
    def __init__(self):
        self.models = {
            'vision_language': 'qwen3-vl:235b-cloud',
            'coding': 'qwen3-coder:480b-cloud',
            'agentic_reasoning': 'glm-4.6:cloud',
            'general_dev': 'gpt-oss:20b-cloud',
            'efficient_coding': 'minimax-m2:cloud'
        }
    
    def analyze_with_context(self, prompt: str, context_files: list, model_type: str = 'coding') -> str:
        """Chain multiple models for complex analysis"""
        
        # First, use coding model to understand the context
        context_analysis = self._call_model(
            self.models['coding'],
            f"Analyze these files and summarize key patterns: {context_files}"
        )
        
        # Then route to specialized model with enriched context
        enhanced_prompt = f"Context analysis: {context_analysis}\n\nUser question: {prompt}"
        return self._call_model(self.models[model_type], enhanced_prompt)
    
    def multimodal_analysis(self, image_path: str, code_snippet: str) -> str:
        """Combine visual and code analysis"""
        with open(image_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        response = ollama.chat(
            model=self.models['vision_language'],
            messages=[{
                'role': 'user',
                'content': [
                    {'type': 'text', 'text': f"Analyze this UI and correlate with code: {code_snippet}"},
                    {'type': 'image', 'source': f"data:image/jpeg;base64,{image_data}"}
                ]
            }]
        )
        return response['message']['content']
    
    def _call_model(self, model: str, prompt: str) -> str:
        response = ollama.chat(model=model, messages=[{'role': 'user', 'content': prompt}])
        return response['message']['content']

# Usage example

result = router.multimodal_analysis('ui_screenshot.png', 'function Button() { return <button>Click</button> }')
print(result)
```
