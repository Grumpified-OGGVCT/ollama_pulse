import ollama
import base64
from typing import List, Dict

class MultiModalDeveloper:
    def __init__(self):
        self.models = {
            'vision': 'qwen3-vl:235b-cloud',
            'reasoning': 'glm-4.6:cloud', 
            'coding': 'qwen3-coder:480b-cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    def analyze_code_with_context(self, code_files: Dict[str, str], question: str) -> str:
        """Use massive context windows to analyze multiple files"""
        context = "\n".join([f"File: {path}\nContent: {content}" 
                           for path, content in code_files.items()])
        
        response = ollama.chat(
            model=self.models['coding'],
            messages=[{
                'role': 'user',
                'content': f"Context: {context}\n\nQuestion: {question}"
            }]
        )
        return response['message']['content']
    
    def visual_to_code(self, image_path: str, requirements: str) -> str:
        """Convert visual designs to functional code"""
        with open(image_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        response = ollama.chat(
            model=self.models['vision'],
            messages=[{
                'role': 'user',
                'content': f"Requirements: {requirements}",
                'images': [image_data]
            }]
        )
        return response['message']['content']

# Practical usage example
dev = MultiModalDeveloper()

# Analyze multiple files at once
files = {
    'auth.py': '... your authentication code ...',
    'database.py': '... your database layer ...', 
    'api.py': '... your API endpoints ...'
}

    files,

)
