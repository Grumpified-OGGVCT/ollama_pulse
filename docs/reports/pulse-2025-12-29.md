### Multi-Model Orchestration Pattern

```python
import ollama
import asyncio
from typing import Dict, Any

class CodePipeline:
    def __init__(self):
        self.models = {
            'analysis': 'qwen3-coder:480b-cloud',
            'agentic': 'glm-4.6:cloud', 
            'efficiency': 'minimax-m2:cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def optimize_codebase(self, code_path: str):
        """Orchestrate multiple models for code optimization"""
        
        # Step 1: Analysis with massive context window
        analysis_prompt = f"""
        Analyze this entire codebase at {code_path} and identify:
        - Performance bottlenecks
        - Code duplication
        - Architectural improvements
        
        Provide specific recommendations with file references.
        """
        
        analysis = await ollama.generate(
            model=self.models['analysis'],
            prompt=analysis_prompt,
            options={'num_ctx': 262000}  # Leverage that huge context!
        )
        
        # Step 2: Agentic planning
        plan_prompt = f"""
        Based on this analysis: {analysis['response']}
        Create a step-by-step refactoring plan prioritizing:
        - Quick wins first
        - Minimal breaking changes
        - Performance impact
        
        Return as JSON with timeline estimates.
        """
        
        plan = await ollama.generate(
            model=self.models['agentic'],
            prompt=plan_prompt
        )
        
        return {
            'analysis': analysis['response'],
            'plan': plan['response']
        }

# Usage
pipeline = CodePipeline()
result = asyncio.run(pipeline.optimize_codebase('/projects/legacy-app'))
```
