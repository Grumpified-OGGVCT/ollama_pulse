Here's a practical Python example for building a multi-agent code review system:

```python
import ollama
import asyncio
from typing import List, Dict

class CodeReviewAgent:
    def __init__(self):
        self.specialists = {
            'security': 'qwen3-coder:480b-cloud',
            'performance': 'glm-4.6:cloud', 
            'architecture': 'gpt-oss:20b-cloud'
        }
    
    async def review_code(self, code: str, review_type: str) -> Dict:
        prompt = f"""
        As a {review_type} specialist, review this code for issues:
        
        {code}
        
        Focus specifically on {review_type} concerns. Provide:
        1. Critical issues found
        2. Recommendations
        3. Code examples for fixes
        """
        
        response = await ollama.generate(
            model=self.specialists[review_type],
            prompt=prompt,
            options={'temperature': 0.1}
        )
        
        return {
            'reviewer': review_type,
            'analysis': response['response'],
            'confidence_score': self._extract_confidence(response)
        }
    
    async def parallel_review(self, code: str) -> List[Dict]:
        tasks = []
        for specialty in self.specialists.keys():
            task = self.review_code(code, specialty)
            tasks.append(task)
        
        return await asyncio.gather(*tasks)

# Usage example
reviewer = CodeReviewAgent()
results = asyncio.run(reviewer.parallel_review("""
def process_user_input(input_data):
    # Example vulnerable code
    eval(input_data['user_code'])
    return {"status": "processed"}
"""))

for result in results:
    print(f"=== {result['reviewer'].upper()} REVIEW ===")
    print(result['analysis'])
```

```python
def create_agent_workflow():
    """Chain models for complex tasks"""
    def workflow(prompt: str):
        # First, use minimax-m2 for fast analysis
        quick_analysis = ollama.generate(
            model='minimax-m2:cloud',
            prompt=f"Quick analysis: {prompt}"
        )
        
        # Then deep dive with specialized model
        if "code" in prompt.lower():
            specialist = 'qwen3-coder:480b-cloud'
        elif "visual" in prompt.lower():
            specialist = 'qwen3-vl:235b-cloud'
        else:
            specialist = 'glm-4.6:cloud'
            
        deep_analysis = ollama.generate(
            model=specialist,
            prompt=f"Based on this analysis: {quick_analysis}\n\nDeep dive: {prompt}"
        )
        
        return deep_analysis
    return workflow
```
