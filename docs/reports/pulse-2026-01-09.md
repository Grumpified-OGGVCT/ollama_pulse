```python
import ollama
import asyncio
from typing import List, Dict

class MultiModelOrchestrator:
    def __init__(self):
        self.models = {
            'vision': 'qwen3-vl:235b-cloud',
            'coding': 'qwen3-coder:480b-cloud', 
            'reasoning': 'glm-4.6:cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def analyze_ui_to_code(self, image_path: str, requirements: str):
        """Convert UI mockup to working code"""
        # Step 1: Visual analysis
        vision_prompt = f"""
        Analyze this UI mockup and extract:
        - Layout structure
        - UI components
        - Interactive elements
        - Styling requirements
        
        Requirements: {requirements}
        """
        
        visual_analysis = await ollama.generate(
            model=self.models['vision'],
            prompt=vision_prompt,
            images=[image_path]
        )
        
        # Step 2: Code generation
        code_prompt = f"""
        Based on this analysis, generate clean, production-ready code:
        
        {visual_analysis.response}
        
        Requirements:
        - Use React with TypeScript
        - Responsive design
        - Accessibility best practices
        """
        
        code_result = await ollama.generate(
            model=self.models['coding'],
            prompt=code_prompt
        )
        
        return {
            'analysis': visual_analysis.response,
            'code': code_result.response
        }

# Usage example
```

```python
def automated_code_review(file_path: str):
    """Multi-aspect code review using specialized models"""
    with open(file_path, 'r') as f:
        code_content = f.read()
    
    review_aspects = {
        'security': "Analyze for security vulnerabilities and best practices",
        'performance': "Identify performance bottlenecks and optimization opportunities", 
        'maintainability': "Assess code readability, structure, and maintainability",
        'testing': "Review test coverage and suggest improvements"
    }
    
    reviews = {}
    for aspect, instruction in review_aspects.items():
        prompt = f"""
        Code to review:
        {code_content}
        
        Focus: {instruction}
        Provide specific, actionable feedback.
        """
        
        response = ollama.generate(
            model='qwen3-coder:480b-cloud' if aspect in ['security', 'performance'] else 'gpt-oss:20b-cloud',
            prompt=prompt
        )
        reviews[aspect] = response.response
    
    # Synthesize with reasoning model
    synthesis_prompt = f"""
    Synthesize these code reviews into a prioritized action plan:
    
    {reviews}
    
    Provide clear next steps for the development team.
    """
    
    final_review = ollama.generate(
        model='glm-4.6:cloud',
        prompt=synthesis_prompt
    )
    
    return final_review.response
```
