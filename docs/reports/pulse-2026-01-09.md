class MultiModalDevAssistant:

        - Modern best practices
        """
        
        return ollama.chat(
            model=self.coder_model,
            messages=[{"role": "user", "content": coder_prompt}]

        }
    
    def build_microservice(self, requirements):
        # Coordinator analyzes requirements and delegates
        coordinator_prompt = f"""
        Analyze these microservice requirements and break them into specialized tasks:
        {requirements}
        
        Assign each task to the most appropriate specialist model.
        Return a structured plan with task assignments.
        """
        
        plan = ollama.chat(model=self.coordinator, messages=[
            {"role": "user", "content": coordinator_prompt}
        ])
        
        results = {}
        for task in self.parse_plan(plan):
            specialist_model = self.specialists[task['type']]
            task_result = ollama.chat(model=specialist_model, messages=[
                {"role": "user", "content": task['prompt']}
            ])
            results[task['type']] = task_result
        
        return self.integrate_results(results)
