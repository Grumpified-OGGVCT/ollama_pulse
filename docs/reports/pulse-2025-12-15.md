```python
import ollama
import base64
from typing import Dict, List

class MultiModelOrchestrator:
    def __init__(self):
        self.specialists = {
            'vision': 'qwen3-vl:235b-cloud',
            'coding': 'qwen3-coder:480b-cloud', 
            'reasoning': 'glm-4.6:cloud',
            'efficiency': 'minimax-m2:cloud'
        }
    
    def analyze_code_screenshot(self, image_path: str, question: str) -> Dict:
        # Convert image to base64 for multimodal input
        with open(image_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        # First, get visual analysis
        vision_prompt = f"Describe this code screenshot and identify key elements: {question}"
        visual_analysis = ollama.chat(
            model=self.specialists['vision'],
            messages=[{
                'role': 'user',
                'content': [
                    {'type': 'text', 'text': vision_prompt},
                    {'type': 'image', 'source': f'data:image/jpeg;base64,{image_data}'}
                ]
            }]
        )
        
        # Then get code analysis
        code_context = f"Based on this visual description: {visual_analysis['message']['content']}"
        code_analysis = ollama.chat(
            model=self.specialists['coding'],
            messages=[{
                'role': 'user', 
                'content': f"{code_context}\n\n{question}"
            }]
        )
        
        return {
            'visual_analysis': visual_analysis['message']['content'],
            'code_analysis': code_analysis['message']['content']
        }

# Usage example
orchestrator = MultiModelOrchestrator()
result = orchestrator.analyze_code_screenshot(
    image_path="bug_screenshot.png",
    question="What's wrong with this UI component and how would you fix it?"
)
```
