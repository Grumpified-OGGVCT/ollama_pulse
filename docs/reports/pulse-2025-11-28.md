    def analyze_design_to_code(self, image_path, requirements):
        # Step 1: Vision analysis
        with open(image_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode()
        
        vision_prompt = f"""
        Analyze this UI design and extract:
        - Layout structure and components
        - Color scheme and styling patterns
        - Interactive elements and their likely behavior
        - Content organization and hierarchy
        
        Design requirements: {requirements}
        """
        
        vision_analysis = ollama.chat(
            model=self.vision_model,
            messages=[{
                "role": "user",
                "content": vision_prompt,
                "images": [image_data]
            }]
        )
        
        # Step 2: Code generation
        code_prompt = f"""
        Based on this design analysis, generate a React component with Tailwind CSS:
        
        {vision_analysis['message']['content']}
        
        Requirements: {requirements}
        
        Return only the clean, production-ready code with proper TypeScript types.
        """
        
        code_generation = ollama.chat(
            model=self.coder_model,
            messages=[{"role": "user", "content": code_prompt}]
        )
        
        return code_generation['message']['content']
