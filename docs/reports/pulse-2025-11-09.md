    async def analyze_architecture_diagram(self, image_path: str, requirements: str):
        """Multi-step analysis using different model strengths"""
        
        # Step 1: Vision model analyzes the diagram
        vision_prompt = f"""
        Analyze this architecture diagram and describe the components, 
        data flow, and technical stack visible. Focus on identifying:
        - Databases and storage systems
        - API endpoints and services
        - Frontend/backend separation
        - Authentication/security layers
        """
        
        vision_response = await ollama.generate(
            model=self.models['vision'],
            prompt=vision_prompt,
            images=[image_path]
        )
        
        # Step 2: Reasoning model plans the implementation
        reasoning_prompt = f"""
        Based on this architecture analysis: {vision_response}
        And these requirements: {requirements}
        
        Create a detailed implementation plan including:
        1. Priority order for components
        2. Technology recommendations
        3. Potential integration challenges
        4. Testing strategy
        """
        
        plan = await ollama.generate(
            model=self.models['reasoning'],
            prompt=reasoning_prompt
        )
        
        # Step 3: Coding specialist generates starter code
        coding_prompt = f"""
        Based on this implementation plan: {plan}
        Generate starter code for the highest priority component.
        Include:
        - Main class/service structure
        - Key function stubs with type hints
        - Basic configuration
        - Example usage
        """
        
        code = await ollama.generate(
            model=self.models['coding'],
            prompt=coding_prompt
        )
        
        return {
            'analysis': vision_response,
            'plan': plan,
            'starter_code': code
        }

# Usage example
