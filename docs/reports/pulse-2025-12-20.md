        Analyze this coding problem: {question}

        if any(keyword in question.lower() for keyword in ['bug', 'fix', 'optimize']):
            tasks.append(self._handle_code_analysis(code_files, question))
            
        results = await asyncio.gather(*tasks)
        return self._synthesize_results(results)
    
    async def _handle_code_analysis(self, code_files: Dict[str, str], question: str):
        """Leverage qwen3-coder's massive context for deep code analysis"""
        context = "\n".join([f"// {filename}\n{content}" for filename, content in code_files.items()])

            Codebase Context (first 200K tokens):
            {context[:200000]}
            
            Question: {question}
            
            Provide specific line numbers, file references, and code fixes.
            """
        )
        return {'type': 'code_analysis', 'result': response}
