    def analyze_ui_and_code(self, screenshot_path, code_files):
        # Convert screenshot to base64 for VL model
        with open(screenshot_path, "rb") as img_file:
            img_b64 = base64.b64encode(img_file.read()).decode()
        
        # Get visual analysis
        vision_prompt = f"Analyze this UI screenshot and identify components, layout, and potential issues."
        vision_response = ollama.chat(
            model=self.vision,
            messages=[{
                "role": "user",
                "content": [
                    {"type": "text", "text": vision_prompt},
                    {"type": "image", "source": f"data:image/jpeg;base64,{img_b64}"}
                ]
            }]
        )
        
        # Feed analysis to coder with relevant code context
        coder_prompt = f"""

        And these relevant code files: {code_files}
        
        Suggest specific improvements and identify mismatches between design and implementation.
        """
        
        return ollama.chat(model=self.coder, messages=[{"role": "user", "content": coder_prompt}])
