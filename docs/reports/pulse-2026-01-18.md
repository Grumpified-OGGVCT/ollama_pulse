import ollama
import os

class CodebaseOracle:
    def __init__(self, model_name="qwen3-coder:480b-cloud"):
        self.model = model_name
        self.context_window = 262000  # Massive context!
    
    def analyze_architecture(self, codebase_path):
        """Analyze entire codebase for architectural patterns"""
        
        # Read all code files recursively
        code_context = ""
        for root, dirs, files in os.walk(codebase_path):
            for file in files:
                if file.endswith(('.py', '.js', '.ts', '.java', '.go')):
                    filepath = os.path.join(root, file)
                    try:
                        with open(filepath, 'r') as f:
                            code_context += f"File: {filepath}\n```\n{f.read()}\n```\n\n"
                    except:
                        pass
        
        # Use the massive context window to analyze everything
        prompt = f"""
        Analyze this codebase for:
        1. Security vulnerabilities
        2. Performance bottlenecks  
        3. Architecture patterns
        4. Integration points
        
        Codebase:
        {code_context[:250000]}  # Stay under context limit
        
        Provide specific recommendations for improvement.
        """
        
        response = ollama.chat(model=self.model, messages=[
            {"role": "user", "content": prompt}
        ])
        
        return response['message']['content']

# Usage
