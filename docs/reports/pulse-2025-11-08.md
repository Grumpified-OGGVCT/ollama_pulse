            'reviewer': 'gpt-oss:20b-cloud',   # Versatile review
            'multimodal': 'qwen3-vl:235b-cloud' # Visual understanding
        }
    
    async def plan_feature(self, requirements: str) -> Dict:
        """Use GLM-4.6 for advanced planning with agentic reasoning"""
        prompt = f"""
        As a senior architect, break down these requirements into implementable tasks:
        {requirements}
        
        Consider: file structure, dependencies, testing strategy, and deployment impact.
        Return JSON with tasks, dependencies, and estimated complexity.
        """
        
        response = await ollama.chat(
            model=self.models['planner'],
            messages=[{'role': 'user', 'content': prompt}]
        )
        return self._parse_plan(response['message']['content'])
    
    async def implement_component(self, spec: Dict, context_files: List[str]) -> str:
        """Use Qwen3-Coder with massive context for implementation"""
        context = "\n".join([f"File: {f}\nContent: {self._read_file(f)}" 
                           for f in context_files])
        
        prompt = f"""
        Based on this architectural context:
        {context}
        
        Implement this component specification:
        {spec}
        
        Ensure consistency with existing patterns and include appropriate tests.
        """
        
        response = await ollama.chat(
            model=self.models['coder'],
            messages=[{'role': 'user', 'content': prompt}]
        )
        return response['message']['content']
    
    async def review_with_visuals(self, code: str, screenshot_path: str) -> Dict:
        """Combine visual and code analysis for comprehensive review"""
        # Analyze the visual interface
        visual_analysis = await ollama.chat(
            model=self.models['multimodal'],
            messages=[{
                'role': 'user', 
                'content': f"Analyze this UI screenshot for consistency with our design system: {screenshot_path}"
            }]
        )
        
        # Review code against visual requirements
        review_prompt = f"""
        Code to review:
        {code}
        
        Visual design requirements from screenshot analysis:
        {visual_analysis['message']['content']}
        
        Provide specific suggestions for alignment.
        """
        
        response = await ollama.chat(
            model=self.models['reviewer'],
            messages=[{'role': 'user', 'content': review_prompt}]
        )
        return {
            'code_review': response['message']['content'],
            'visual_alignment': visual_analysis['message']['content']
        }
