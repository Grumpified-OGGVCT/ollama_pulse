import asyncio
from typing import Dict, Any

class MultiModelDeveloper:
    def __init__(self):
        self.models = {
            'reasoning': 'glm-4.6:cloud',
            'coding': 'qwen3-coder:480b-cloud', 
            'vision': 'qwen3-vl:235b-cloud',
            'general': 'gpt-oss:20b-cloud'
        }
    
    async def orchestrate_development_task(self, task_description: str, visual_context: str = None):
        """Orchestrate a complex development task across specialized models"""
        
        # Step 1: Use reasoning model to break down the task
        reasoning_prompt = f"""
        Break this development task into subtasks and recommend which models should handle each:
        Task: {task_description}
        
        Return JSON with: {{"subtasks": [{{"description": "", "model": "", "dependencies": []}}]}}
        """
        
        plan = await self.call_model('reasoning', reasoning_prompt)
        
        # Step 2: Execute subtasks based on dependencies
        results = {}
        for subtask in plan['subtasks']:
            if all(dep in results for dep in subtask['dependencies']):
                context = "\n".join([results[dep] for dep in subtask['dependencies']])
                prompt = f"Context: {context}\n\nSubtask: {subtask['description']}"
                results[subtask['description']] = await self.call_model(subtask['model'], prompt)
        
        return results
    
    async def call_model(self, model_type: str, prompt: str):
        response = ollama.chat(model=self.models[model_type], messages=[
            {'role': 'user', 'content': prompt}
        ])
        return response['message']['content']

# Usage example
async def main():
    dev_assistant = MultiModelDeveloper()
    
    result = await dev_assistant.orchestrate_development_task(
        "Create a React component that displays user analytics data from a provided CSV file"
