### Agentic Workflow with Long Context

```python
class CodebaseAnalyzer:
    def __init__(self):
        self.model = 'glm-4.6:cloud'
        self.context_window = 200000
        
    def load_codebase(self, directory):
        # Load and chunk entire codebase
        code_files = []
        for root, dirs, files in os.walk(directory):
            for file in files:
                if file.endswith(('.py', '.js', '.ts', '.java')):
                    path = os.path.join(root, file)
                    with open(path, 'r') as f:
                        code_files.append(f"{path}:\n{f.read()}")
        
        return "\n\n".join(code_files)
    
    def analyze_refactor(self, codebase_content, refactor_task):
        prompt = f"""
        Codebase Context ({len(codebase_content)} chars):
        {codebase_content[:self.context_window]}
        
        Refactor Task: {refactor_task}
        
        Provide a step-by-step refactoring plan with specific file changes.
        """
        
        response = ollama.chat(
            model=self.model,
            messages=[{'role': 'user', 'content': prompt}]
        )
        
        return response['message']['content']

# Usage
analyzer = CodebaseAnalyzer()
codebase = analyzer.load_codebase('./src')
refactor_plan = analyzer.analyze_refactor(
    codebase, 
    'Convert all class components to functional components with hooks'
)
```
