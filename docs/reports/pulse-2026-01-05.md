class MultiModalDevAgent:

            vision_analysis += response['message']['content'] + "\n\n"

        Identify inconsistencies between the architecture and implementation.

        """
        
        return ollama.chat(
            model=self.models['coding'],
            messages=[{'role': 'user', 'content': analysis_prompt}]
