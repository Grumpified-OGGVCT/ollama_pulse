        vision_response = ollama.chat(

            }]
        )
        
        # Get code fixes from coding specialist
        code_response = ollama.chat(
            model=self.coder_model,
            messages=[{
                "role": "user",
                "content": f"Based on this UI analysis: {vision_response['message']['content']}. Provide specific HTML/CSS/JS fixes."
            }]
        )
        
        return {
            "analysis": vision_response['message']['content'],
            "fixes": code_response['message']['content']
        }
