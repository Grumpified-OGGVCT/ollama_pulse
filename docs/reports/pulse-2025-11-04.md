    async def analyze_ui_design(self, image_path: str, requirements: str) -> Dict:
        """Use vision model to analyze UI and generate specifications"""
        
        # Step 1: Visual analysis
        vision_prompt = f"""

        - Component hierarchy
        - Layout structure

        """
        
        vision_response = await ollama.generate(
            model=self.models['vision'],

            images=[image_path]
        )

        coding_prompt = f"""
        Based on this analysis, generate React component specifications:
        
        {vision_response}
        
        Include:
        - PropTypes definitions
        - State management requirements
        - Event handlers needed
        """
        
        component_specs = await ollama.generate(
            model=self.models['coding'],
            prompt=coding_prompt
        )
        
        # Step 3: Architecture recommendations
        reasoning_prompt = f"""
        Given these components, recommend an optimal architecture:
        
        {component_specs}
        
        Consider:
        - State management strategy
        - Performance optimizations
        - Testing approach
        """
        
        architecture = await ollama.generate(
            model=self.models['reasoning'],
            prompt=reasoning_prompt
        )
        
        return {
            'analysis': vision_response,
            'components': component_specs,
            'architecture': architecture
        }
