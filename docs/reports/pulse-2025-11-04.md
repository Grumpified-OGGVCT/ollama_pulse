            'general': 'gpt-oss:20b-cloud'
        }
    
    async def analyze_ui_component(self, image_path: str, code_snippet: str):
        """Use multiple models to analyze a UI component"""
        
        # Vision model analyzes the design
        vision_prompt = f"Analyze this UI design and extract key components, colors, spacing, and layout principles."
        vision_analysis = await ollama.generate(
            model=self.models['vision'],
            prompt=vision_prompt,
            images=[image_path]
        )
        
        # Coding specialist suggests implementation
        coder_prompt = f"""
        Based on this design analysis: {vision_analysis}
        And this existing code structure: {code_snippet}
        
        Suggest the best implementation approach considering:
        - Component reusability
        - Performance optimizations
        - Accessibility best practices
        """
        
        implementation_plan = await ollama.generate(
            model=self.models['coding'],
            prompt=coder_prompt
        )
        
        # Reasoning model evaluates the approach
        evaluation_prompt = f"""
        Evaluate this implementation plan: {implementation_plan}
        
        Consider:
        - Potential technical debt
        - Scalability concerns
        - Maintenance implications
        """
        
        final_evaluation = await ollama.generate(
            model=self.models['reasoning'],
            prompt=evaluation_prompt
        )
        
        return {
            'design_analysis': vision_analysis,
            'implementation_plan': implementation_plan,
            'technical_evaluation': final_evaluation
        }
