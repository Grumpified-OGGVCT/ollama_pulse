        self.coder_model = "qwen3-coder:480b-cloud"
        self.agent_model = "glm-4.6:cloud"
    
    def analyze_code_with_ui(self, code: str, screenshot_path: str) -> Dict:
        # Convert image to base64 for multimodal input
        with open(screenshot_path, "rb") as image_file:
            image_data = base64.b64encode(image_file.read()).decode('utf-8')
        
        prompt = f"""
        Analyze this code and corresponding UI:
        CODE: {code}
        
        Provide specific suggestions for:
        1. UI-code consistency
        2. Accessibility improvements  
        3. Performance optimizations
        4. Cross-browser compatibility
        """
        
        response = ollama.generate(
            model=self.vl_model,
            prompt=prompt,
            images=[image_data]
        )
        
        return self._parse_analysis(response['response'])
    
    def polyglot_code_review(self, files: List[Dict]) -> str:
        """Review code across multiple languages in context"""
        context = "\n".join([f"LANGUAGE: {f['lang']}\nCODE: {f['content']}" for f in files])
        
        response = ollama.generate(
            model=self.coder_model,
            prompt=f"""Review this multi-language codebase for consistency, security, and best practices:
            
            {context}
            
            Focus on:
            - API consistency across services
            - Shared security patterns
            - Performance implications
            - Deployment coordination
            """
        )
        
        return response['response']
